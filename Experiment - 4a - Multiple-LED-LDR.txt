import RPi.GPIO as GPIO
import time

GPIO.setwarnings(False)
GPIO.cleanup()
GPIO.setmode(GPIO.BCM)

LDR_PIN = 25

led_pins=[21,17]

for pin in led_pins:
    GPIO.setup(pin, GPIO.OUT)
    
GPIO.setup(LDR_PIN,GPIO.IN)

while True:
    if (GPIO.input(LDR_PIN)==1):
        for i in range(2):
            GPIO.output(led_pins[i],True)
    else:
        for i in range(2):
         GPIO.output(led_pins[i],False)
